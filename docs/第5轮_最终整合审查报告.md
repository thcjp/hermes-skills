# ============ 在文件顶部添加字段映射常量 ============
# TRACE五维度到scores表字段的正确映射
# 由于历史原因scores表字段名与TRACE维度名不一致，使用此映射保证语义正确
TRACE_FIELD_MAPPING = {
    'trust': 'debranding_score',           # Trust分存入debranding_score
    'reliability': 'quality_score',          # Reliability分存入quality_score
    'adaptability': 'practicality_score',    # Adaptability分存入practicality_score

def save_trace_score(skill_id, checks, static_scores, llm_result=None):
    """保存TRACE评分到数据库
    
    注意：scores表字段名与TRACE维度名不一致，使用TRACE_FIELD_MAPPING保证语义正确。
    differentiation_score和compliance_score不再重复存储trust_score，

    if llm_result:
        trace = llm_result.get('trace_scores', {})
        trust_score = max(static_scores['trust_static'], trace.get('trust', {}).get('score', 0))
        reliability_score = trace.get('reliability', {}).get('score', 0)
        adaptability_score = trace.get('adaptability', {}).get('score', 0)
        convention_score = max(static_scores['convention_static'], trace.get('convention', {}).get('score', 0))
        effectiveness_score = trace.get('effectiveness', {}).get('score', 0)
        total = round(trust_score + reliability_score + adaptability_score + convention_score + effectiveness_score, 1)
        grade = llm_result.get('quality_grade', 'D')
        notes = json.dumps({
            'issues': checks['issues'],
            'top_3': llm_result.get('top_3_issues', []),

            'static_scores': static_scores,

            'evaluated_at': now,
            'field_mapping': TRACE_FIELD_MAPPING,  # 记录映射关系供后续迁移
        }, ensure_ascii=False)
        is_pass = 1 if total >= TRACE_PASS_THRESHOLD else 0  # 使用统一阈值
    else:
        trust_score = static_scores['trust_static']
        reliability_score = 0
        adaptability_score = 0
        convention_score = static_scores['convention_static']

        total = round(trust_score + convention_score, 1)

        notes = json.dumps({
            'issues': checks['issues'],

### 6.2 关键风险链分析

**风险链1：数据迁移风险**
```
scores表迁移 → 历史评分数据可能错乱 → 门控判断异常
    → 高价值skill被误阻止 → 收入损失
```
**规避**：迁移前完整备份 + 迁移后运行验证脚本 + 保留旧表只读1个月

**风险链2：支付合规风险**
```
自行收款 → 需要支付业务许可证 → 无证经营 → 法律风险
```
**规避**：使用第三方支付聚合服务（Ping++/Stripe），由平台代收代付

**风险链3：定价调整风险**
```
定价调整 → 现有用户不满 → 流失 → 收入下降
```
**规避**：先对新skill应用新定价，老skill保持原价3个月观察期

**风险链4：批量操作风险**
```
批量修复 → 引入新bug → 1858个SKILL.md被破坏 → 库存损毁
```
**规避**：强制dry-run + 备份 + 回滚机制 + 修复后自动验证

| 场景 | 触发条件 | 回滚动作 | 预计恢复时间 |

### 8.2 核心结论

1. **当前系统不具备赚钱能力**：变现基础设施完全缺失（支付/计量/收入），即使修复所有代码问题，也只是"有了更好的商品"，但"无法卖出"。

2. **数据可信度是第一优先**：scores表字段错位是"根因性"问题，它污染了从评分到定价到门控的全链路。不修复此问题，所有下游改进都是建立在错误数据上。

3. **商品库存安全是底线**：1858个SKILL.md可被批量覆写且无回滚，这是"一次事故归零"的风险。必须在任何大规模操作前建立备份+回滚机制。

4. **定价策略需要根本性转变**：从成本导向转向价值导向，从单一per_call转向多元化计费（per_call + monthly + annual + freemium）。

5. **工程实践是可持续发展的基础**：零测试+无CI+无回滚构成了"缺陷直通生产"的高速通道。即使短期赚钱，一次事故就可能归零。
