### 基础用法

```python
# 知识提取基础
from datetime import datetime

def extract_knowledge(conversation):
    knowledge = {
        "date": datetime.now().isoformat(),
        "topics": [],
        "decisions": [],
        "action_items": [],
        "references": []
    }
    # 简化提取逻辑
    for line in conversation.split("\n"):
        if "决定" in line:
            knowledge["decisions"].append(line)
        elif "需要" in line or "应该" in line:
            knowledge["action_items"].append(line)
    return knowledge

conv = "我们决定使用Kubernetes\n需要配置CI/CD\n应该下周完成"
result = extract_knowledge(conv)
print(f"决策: {len(result['decisions'])} 条")
print(f"行动项: {len(result['action_items'])} 条")
```
