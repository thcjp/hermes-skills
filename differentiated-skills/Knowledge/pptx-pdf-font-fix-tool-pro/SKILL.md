### 基础用法

```python
# 企业级PPT字体修复引擎（PRO）
import os
import json
from typing import List, Dict, Optional, Set
from pathlib import Path
from pptx import Presentation
from pptx.util import Pt
from dataclasses import dataclass, field
# ...
@dataclass
class FontIssue:
    slide_num: int
    shape_name: str
    font_name: str
    issue_type: str  # missing, embedded, substituted
    fix_applied: str = ""
# ...
class PPTFontFixEngine:
    FONT_MAP = {
        "宋体": "SimSun",
        "黑体": "SimHei",
        "微软雅黑": "Microsoft YaHei",
        "楷体": "KaiTi",
        "仿宋": "FangSong",
        "Arial": "Arial",
        "Times New Roman": "Times New Roman",
    }
# ...
    SAFE_FONTS = ["SimSun", "SimHei", "Microsoft YaHei", "Arial"]
# ...
    def __init__(self):
        self.issues: List[FontIssue] = []
        self.fixed_fonts: Dict[str, str] = {}
# ...
    def analyze(self, pptx_path: str) -> List[FontIssue]:
        """分析字体问题（PRO 专属）"""
        prs = Presentation(pptx_path)
        self.issues = []
        all_fonts: Set[str] = set()
        for i, slide in enumerate(prs.slides, 1):
            for shape in slide.shapes:
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        for run in para.runs:
                            if run.font.name:
                                font = run.font.name
                                all_fonts.add(font)
                                if font not in self.SAFE_FONTS:
                                    self.issues.append(FontIssue(
                                        slide_num=i,
                                        shape_name=shape.name,
                                        font_name=font,
                                        issue_type="potentially_unsafe"
                                    ))
        return self.issues
# ...
    def fix_fonts(self, pptx_path: str, output_path: str,
                 font_mapping: Dict[str, str] = None) -> dict:
        """修复字体（PRO 专属：自动替换）"""
        if font_mapping is None:
            font_mapping = self._get_default_mapping()
        prs = Presentation(pptx_path)
        replaced = {}
        for slide in prs.slides:
            for shape in slide.shapes:
                if shape.has_text_frame:
                    for para in shape.text_frame.paragraphs:
                        for run in para.runs:
                            if run.font.name and run.font.name in font_mapping:
                                old = run.font.name
                                new = font_mapping[old]
                                run.font.name = new
                                replaced[old] = new
        prs.save(output_path)
        self.fixed_fonts = replaced
        return {
            "output": output_path,
            "fonts_replaced": len(replaced),
            "mapping": replaced
        }
# ...
    def convert_to_pdf(self, pptx_path: str, output_pdf: str) -> dict:
        """转为PDF（PRO 专属：字体安全）"""
        import subprocess
        fixed_pptx = output_pdf.replace(".pdf", "_fixed.pptx")
        self.fix_fonts(pptx_path, fixed_pptx)
        result = subprocess.run([
            "libreoffice", "--headless", "--convert-to", "pdf",
            "--outdir", str(Path(output_pdf).parent), fixed_pptx
        ], capture_output=True, text=True)
        return {
            "status": "success" if result.returncode == 0 else "failed",
            "output": output_pdf,
            "error": result.stderr if result.returncode != 0 else None
        }
# ...
    def batch_fix(self, pptx_files: List[str],
                 output_dir: str = "./output") -> List[dict]:
        """批量修复（PRO 专属）"""
        Path(output_dir).mkdir(exist_ok=True)
        results = []
        for pptx_file in pptx_files:
            output_path = str(Path(output_dir) / Path(pptx_file).name)
            result = self.fix_fonts(pptx_file, output_path)
            result["source"] = pptx_file
            results.append(result)
        return results
# ...
    def generate_report(self, output_path: str):
        """生成修复报告（PRO 专属）"""
        report = {
            "total_issues": len(self.issues),
            "issues_by_type": {},
            "fixed_fonts": self.fixed_fonts,
            "details": [i.__dict__ for i in self.issues]
        }
        for issue in self.issues:
            report["issues_by_type"][issue.issue_type] = \
                report["issues_by_type"].get(issue.issue_type, 0) + 1
        with open(output_path, "w", encoding="utf-8") as f:
            json.dump(report, f, ensure_ascii=False, indent=2)
# ...
    def _get_default_mapping(self) -> Dict[str, str]:
        mapping = {}
        for cn_name, en_name in self.FONT_MAP.items():
            if cn_name != en_name:
                mapping[cn_name] = en_name
        return mapping
# ...
engine = PPTFontFixEngine()
issues = engine.analyze("presentation.pptx")
print(f"发现 {len(issues)} 个字体问题")
result = engine.fix_fonts("presentation.pptx", "fixed.pptx")
print(f"替换了 {result['fonts_replaced']} 种字体")
engine.generate_report("font_report.json")
```

## 常见问题

### Q1: 批量处理时遇到内存不足？

A: 专业版支持分批处理，建议减小batch_size参数，或增加并行度但减少每批文件数量.
### Q2: 如何配置自动重试？

A: 在配置文件中设置retry_attempts和retry_delay参数。专业版支持指数退避重试策略.
### Q3: 如何监控处理进度？

A: 专业版内置进度追踪功能，通过回调或轮询方式获取实时处理状态。可配置webhook通知.
### Q4: 如何与现有系统集成？

A: 专业版提供完整的API接口和配置文件，支持CI/CD集成、定时任务和webhook回调.
