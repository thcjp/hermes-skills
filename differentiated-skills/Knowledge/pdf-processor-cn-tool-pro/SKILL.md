### 基础用法

```python
# 企业级中文PDF处理引擎（PRO）
import os
import json
import re
from typing import List, Dict, Optional
from pathlib import Path
from dataclasses import dataclass, field
from pypdf import PdfReader
import pdfplumber
# ...
@dataclass
class CNPDFPage:
    page_num: int
    text: str
    tables: List[List[List[str]]] = field(default_factory=list)
    layout: str = "single"  # single, double
    confidence: float = 0.0
# ...
class CNPDFProcessor:
    def __init__(self):
        self.pages: List[CNPDFPage] = []
        self.ocr_enabled = False
# ...
    def process(self, pdf_path: str, ocr: bool = False) -> dict:
        """完整处理中文PDF（PRO 专属）"""
        self.ocr_enabled = ocr
        reader = PdfReader(pdf_path)
        with pdfplumber.open(pdf_path) as pdf:
            for i, page in enumerate(pdf.pages):
                cn_page = self._process_page(page, i, reader)
                self.pages.append(cn_page)
        return self._generate_summary()
# ...
    def extract_text(self, pdf_path: str) -> str:
        """提取中文文本（PRO 专属：版面保持）"""
        text_parts = []
        with pdfplumber.open(pdf_path) as pdf:
            for page in pdf.pages:
                text = page.extract_text() or ""
                text = self._fix_cn_text(text)
                text_parts.append(text)
        return NL.join(text_parts)
# ...
    def extract_tables(self, pdf_path: str,
                      export_excel: bool = True) -> List[dict]:
        """提取中文表格（PRO 专属：自动导出）"""
        import pandas as pd
        all_tables = []
        with pdfplumber.open(pdf_path) as pdf:
            for i, page in enumerate(pdf.pages):
                tables = page.extract_tables()
                for j, table in enumerate(tables):
                    if table and len(table) > 1:
                        df = pd.DataFrame(table[1:], columns=table[0])
                        table_data = {
                            "page": i + 1,
                            "table_index": j + 1,
                            "rows": len(df),
                            "columns": len(df.columns),
                            "data": df.to_dict(orient="records")
                        }
                        all_tables.append(table_data)
        if export_excel and all_tables:
            output = Path(pdf_path).stem + "_tables.xlsx"
            with pd.ExcelWriter(output) as writer:
                for idx, t in enumerate(all_tables):
                    df = pd.DataFrame(t["data"])
                    df.to_excel(writer, sheet_name=f"表{idx+1}", index=False)
        return all_tables
# ...
    def ocr_chinese(self, pdf_path: str,
                   lang: str = "chi_sim") -> List[str]:
        """中文OCR识别（PRO 专属）"""
        try:
            import pytesseract
            from pdf2image import convert_from_path
            results = []
            images = convert_from_path(pdf_path)
            for i, img in enumerate(images):
                text = pytesseract.image_to_string(img, lang=lang)
                results.append(f"--- 第{i+1}页 ---{NL}{text}")
            return results
        except ImportError:
            return ["OCR依赖未安装: pip install pytesseract pdf2image"]
# ...
    def smart_chunk(self, pdf_path: str,
                   chunk_size: int = 500) -> List[dict]:
        """智能分块（PRO 专属：按语义切分）"""
        text = self.extract_text(pdf_path)
        chunks = []
        paragraphs = text.split(NL + NL)
        current = ""
        for para in paragraphs:
            if len(current) + len(para) > chunk_size and current:
                chunks.append({"content": current, "length": len(current)})
                current = para
            else:
                current = current + NL + NL + para if current else para
        if current:
            chunks.append({"content": current, "length": len(current)})
        return chunks
# ...
    def batch_process(self, pdf_paths: List[str],
                     output_dir: str = "./output") -> List[dict]:
        """批量处理（PRO 专属）"""
        Path(output_dir).mkdir(exist_ok=True)
        results = []
        for pdf_path in pdf_paths:
            result = self.process(pdf_path)
            result["file"] = pdf_path
            results.append(result)
            # 导出文本
            text = self.extract_text(pdf_path)
            output_file = Path(output_dir) / (Path(pdf_path).stem + ".txt")
            output_file.write_text(text, encoding="utf-8")
        return results
# ...
    def _process_page(self, page, page_idx, reader):
        text = page.extract_text() or ""
        text = self._fix_cn_text(text)
        tables = page.extract_tables() or []
        layout = self._detect_layout(page)
        return CNPDFPage(
            page_num=page_idx + 1,
            text=text,
            tables=tables,
            layout=layout,
            confidence=0.9 if text else 0.0
        )
# ...
    def _fix_cn_text(self, text: str) -> str:
        text = text.replace("\u3000", " ")
        text = re.sub(r"\n{3,}", NL + NL, text)
        return text.strip()
# ...
    def _detect_layout(self, page) -> str:
        words = page.extract_words() or []
        if len(words) > 10:
            x_coords = [w["x0"] for w in words]
            mid_x = (page.width or 612) / 2
            left_count = sum(1 for x in x_coords if x < mid_x)
            right_count = len(x_coords) - left_count
            if left_count > 5 and right_count > 5:
                return "double"
        return "single"
# ...
    def _generate_summary(self) -> dict:
        return {
            "total_pages": len(self.pages),
            "total_text_length": sum(len(p.text) for p in self.pages),
            "total_tables": sum(len(p.tables) for p in self.pages),
            "double_column_pages": sum(1 for p in self.pages if p.layout == "double"),
            "avg_confidence": sum(p.confidence for p in self.pages) / len(self.pages) if self.pages else 0
        }
# ...
proc = CNPDFProcessor()
summary = proc.process("chinese_report.pdf")
print(f"页数: {summary['total_pages']}")
print(f"文本长度: {summary['total_text_length']}")
print(f"表格数: {summary['total_tables']}")
print(f"双栏页: {summary['double_column_pages']}")
```

## 常见问题

### Q1: 批量处理时遇到内存不足？

A: 专业版支持分批处理，建议减小batch_size参数，或增加并行度但减少每批文件数量.
### Q2: 如何配置自动重试？

A: 在配置文件中设置retry_attempts和retry_delay参数。专业版支持指数退避重试策略.
### Q3: 如何监控处理进度？

A: 专业版内置进度追踪功能，通过回调或轮询方式获取实时处理状态。可配置webhook通知.
### Q4: 如何与现有系统集成？

A: 专业版提供完整的API接口和配置文件，支持CI/CD集成、定时任务和webhook回调.
