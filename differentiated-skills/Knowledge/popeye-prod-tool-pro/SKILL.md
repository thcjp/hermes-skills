### 基础用法

```python
# 企业级生产力增强引擎（PRO）
import json
from typing import List, Dict, Optional
from datetime import datetime, timedelta
from dataclasses import dataclass, field
# ...
@dataclass
class Task:
    id: str
    title: str
    priority: str = "medium"
    status: str = "pending"
    due: Optional[str] = None
    tags: List[str] = field(default_factory=list)
    subtasks: List[str] = field(default_factory=list)
    time_spent: float = 0.0
    created: str = ""
    completed: Optional[str] = None
# ...
@dataclass
class TimeEntry:
    task_id: str
    start: str
    end: str
    duration: float
    category: str
# ...
@dataclass
class Workflow:
    name: str
    steps: List[dict] = field(default_factory=list)
    trigger: str = "manual"
    enabled: bool = True
# ...
class ProductivityEngine:
    def __init__(self):
        self.tasks: Dict[str, Task] = {}
        self.time_entries: List[TimeEntry] = []
        self.workflows: Dict[str, Workflow] = {}
        self.habits: Dict[str, dict] = {}
        self.focus_sessions: List[dict] = []
# ...
    def create_task(self, title: str, priority: str = "medium",
                   due: str = None, tags: List[str] = None,
                   subtasks: List[str] = None) -> Task:
        """创建任务（PRO 专属：子任务+标签）"""
        task_id = f"task_{len(self.tasks) + 1}"
        task = Task(
            id=task_id, title=title, priority=priority,
            due=due, tags=tags or [], subtasks=subtasks or [],
            created=datetime.now().isoformat()
        )
        self.tasks[task_id] = task
        return task
# ...
    def track_time(self, task_id: str, duration: float,
                  category: str = "work") -> TimeEntry:
        """时间追踪（PRO 专属）"""
        entry = TimeEntry(
            task_id=task_id,
            start=datetime.now().isoformat(),
            end=(datetime.now() + timedelta(hours=duration)).isoformat(),
            duration=duration, category=category
        )
        self.time_entries.append(entry)
        if task_id in self.tasks:
            self.tasks[task_id].time_spent += duration
        return entry
# ...
    def start_focus_session(self, duration: int = 25,
                          task_id: str = None) -> dict:
        """专注模式（PRO 专属：番茄钟）"""
        session = {
            "start": datetime.now().isoformat(),
            "duration_minutes": duration,
            "task_id": task_id,
            "type": "pomodoro"
        }
        self.focus_sessions.append(session)
        return session
# ...
    def create_workflow(self, name: str, steps: List[dict],
                       trigger: str = "manual") -> Workflow:
        """创建自动化工作流（PRO 专属）"""
        wf = Workflow(name=name, steps=steps, trigger=trigger)
        self.workflows[name] = wf
        return wf
# ...
    def execute_workflow(self, name: str) -> dict:
        """执行工作流（PRO 专属）"""
        wf = self.workflows.get(name)
        if not wf or not wf.enabled:
            return {"error": "工作流不存在或已禁用"}
        results = []
        for step in wf.steps:
            results.append({
                "step": step.get("name", ""),
                "action": step.get("action", ""),
                "status": "executed"
            })
        return {"workflow": name, "steps_executed": len(results), "results": results}
# ...
    def track_habit(self, name: str, target: int = 1) -> dict:
        """习惯追踪（PRO 专属）"""
        today = datetime.now().strftime("%Y-%m-%d")
        if name not in self.habits:
            self.habits[name] = {"target": target, "history": {}}
        self.habits[name]["history"][today] = self.habits[name]["history"].get(today, 0) + 1
        return {"habit": name, "today": self.habits[name]["history"][today]}
# ...
    def analyze_productivity(self, period: str = "week") -> dict:
        """效率分析（PRO 专属）"""
        now = datetime.now()
        if period == "week":
            start = now - timedelta(days=7)
        elif period == "month":
            start = now - timedelta(days=30)
        else:
            start = now - timedelta(days=1)
        period_entries = [e for e in self.time_entries
                         if datetime.fromisoformat(e.start) >= start]
        total_time = sum(e.duration for e in period_entries)
        categories = {}
        for e in period_entries:
            categories[e.category] = categories.get(e.category, 0) + e.duration
        completed = sum(1 for t in self.tasks.values()
                       if t.status == "completed" and t.completed
                       and datetime.fromisoformat(t.completed) >= start)
        return {
            "period": period,
            "total_hours": round(total_time, 1),
            "tasks_completed": completed,
            "focus_sessions": len([s for s in self.focus_sessions
                                  if datetime.fromisoformat(s["start"]) >= start]),
            "time_by_category": {k: round(v, 1) for k, v in categories.items()},
            "avg_daily_hours": round(total_time / 7, 1) if period == "week" else round(total_time, 1)
        }
# ...
    def generate_daily_report(self) -> str:
        """生成日报（PRO 专属）"""
        today = datetime.now().strftime("%Y-%m-%d")
        completed = [t for t in self.tasks.values()
                    if t.status == "completed" and t.completed
                    and t.completed.startswith(today)]
        pending = [t for t in self.tasks.values() if t.status == "pending"]
        report = f"# 日报 - {today}{NL}{NL}"
        report += f"## 已完成 ({len(completed)}){NL}"
        for t in completed:
            report += f"- [x] {t.title}{NL}"
        report += f"{NL}## 待办 ({len(pending)}){NL}"
        for t in pending[:5]:
            report += f"- [ ] {t.title} [{t.priority}]{NL}"
        analysis = self.analyze_productivity("day")
        report += f"{NL}## 时间统计{NL}"
        report += f"- 总时长: {analysis['total_hours']}小时{NL}"
        report += f"- 专注次数: {analysis['focus_sessions']}{NL}"
        return report
# ...
    def batch_create_tasks(self, task_list: List[dict]) -> List[str]:
        """批量创建任务（PRO 专属）"""
        ids = []
        for item in task_list:
            task = self.create_task(item["title"], item.get("priority", "medium"),
                                   item.get("due"), item.get("tags"))
            ids.append(task.id)
        return ids
# ...
engine = ProductivityEngine()
engine.create_task("完成项目报告", "high", "今天", ["工作", "报告"])
engine.create_task("团队会议", "medium", "明天", ["会议"])
engine.track_time("task_1", 2.5, "deep_work")
engine.start_focus_session(25, "task_1")
engine.create_workflow("日报自动化", [
    {"name": "收集", "action": "汇总今日任务"},
    {"name": "分析", "action": "计算时间统计"},
    {"name": "生成", "action": "输出日报"}
])
print(engine.generate_daily_report())
```

## 常见问题

### Q1: 批量处理时遇到内存不足？

A: 专业版支持分批处理，建议减小batch_size参数，或增加并行度但减少每批文件数量。

### Q2: 如何配置自动重试？

A: 在配置文件中设置retry_attempts和retry_delay参数。专业版支持指数退避重试策略。

### Q3: 如何监控处理进度？

A: 专业版内置进度追踪功能，通过回调或轮询方式获取实时处理状态。可配置webhook通知。

### Q4: 如何与现有系统集成？

A: 专业版提供完整的API接口和配置文件，支持CI/CD集成、定时任务和webhook回调。
