### 基础用法

```python
# 飞书文档内容追加
import subprocess

def append_content(doc_token, content):
    result = subprocess.run([
        "node", "index.js",
        "--action", "append",
        "--token", doc_token,
        "--content", content
    ], capture_output=True, text=True)
    return result.stdout

append_content("doccnXXXX", "# 新章节\n这是追加的内容")
print("内容已追加")
```
